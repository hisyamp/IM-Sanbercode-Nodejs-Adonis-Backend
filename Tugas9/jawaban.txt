1. CREATE DATABASE myshop
2. CREATE TABLE `users` (
  `id` int NOT NULL PRIMARY KEY,
  `name` varchar(255),
  `email` varchar(255),
  `password` varchar(255)
)
CREATE TABLE `items` (
  `id` int NOT NULL PRIMARY KEY,
  `name` varchar(255),
  `description` varchar(255),
  `price` int,
  `stock` int,
  `category_id` int
)
